jQuery Selectors:

	ID Selector Basics:
		$("#(idNameHere)").css("(propertyToChange)", "(newProperty)");

	Class Selector Basics:
		$(."(classNameHere)")

	Element Selector Basics:
		$("(elementNameHere)")

	Multiple Selector Basics:
		$("(elementOne), (elementTwo), (#classOne)")

		$((id,class,element)).css({
			"property": "attribute",
			"property": "attribute",
			"property": "attribute",
		}).
